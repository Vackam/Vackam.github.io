Binary Tree Level Order Traversal

서론

| https://neetcode.io/problems/level-order-traversal-of-binary-tree/

sibling 들을 모아서 vector 형태로 작성해보라.

푸는 방법

bfs로 풀 수 있을 거 같다는 생각이 들었다. 하게 되면 같은 층에 있는 것들을 효율적으로 검색할 수 있다고 생각했다.

처음 풀이에선 q의 크기만큼 해서 sibling들을 구분해야 한다고 생각했는데, 그것이 안될 거 같아서 queue 자체를 두 개를 놓아서 풀었다.

다음은 초기 코드이다.

class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> result;
        if (root == nullptr) return result;
        // 이거 그걸로 풀 수 있지 않나? bfs
        // 근데 개수를 알 수가 없으니까, 개수 for문은 못돌겠고
        // 걍 큐 두 개로 돌아야 겠다.
        queue<TreeNode*> q1;
        queue<TreeNode*> q2;

        bool is_q1_now = true;

        auto push_protocol = [](TreeNode* &root, queue<TreeNode*> &push_q) {
            if (root == nullptr) return;
            if (root->left != nullptr) push_q.push(root->left);
            if (root->right != nullptr) push_q.push(root->right);
        };

        auto copy_result_protocol = [](queue<TreeNode*> target_q) {
            vector<int> result;
            while (!target_q.empty()) {
                auto temp = target_q.front();
                target_q.pop();
                result.push_back(temp->val);
            }

            return result;
        };

        q1.push(root);

        while (!q1.empty() || !q2.empty()) {
            if (is_q1_now) {
                result.push_back(copy_result_protocol(q1));

                while (!q1.empty()) {
                    auto temp = q1.front();
                    q1.pop();
                    push_protocol(temp, q2);
                }
            }
            else {
                result.push_back(copy_result_protocol(q2));

                while (!q2.empty()) {
                    auto temp = q2.front();
                    q2.pop();
                    push_protocol(temp, q1);
                }
            }

            is_q1_now = !is_q1_now;
        }

        return result;
    }
};

각 큐가 시작할 때마다 vector를 만들어서 result에 넣어 주도록 하였다.

하지만, 이렇게 풀지 않고 queue 하나로도 충분하다.

다음은 queue 하나로 푼 방식이다.

class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        if (root == NULL) return {};
        vector<vector<int>> res;

        queue<TreeNode*> q;
        q.push(root);

        while (!q.empty()) {
            vector<int> level;
            int size = q.size();

            for (int i=q.size(); i>0; --i) {
                TreeNode* node = q.front();
                q.pop();
                if (node) {
                    level.push_back(node->val);
                    q.push(node->left);
                    q.push(node->right);
                }
            }

            if (!level.empty()) {
                res.push_back(level);
            }
        }    
        return res;
    }
};
