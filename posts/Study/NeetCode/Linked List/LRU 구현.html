LRU Cache를 구현하시오

푸는 방법

첫 생각에는 대충 스택과 큐 쓰면 되겠다고 판단했는데, 이렇게 될 경우 중간에 있는 놈을 get 했을 때 업데이트를 할 수가 없다.

그래서 링크드리스트를 써야 되는 구나 생각했는데, 그걸 하나하나 구현하는 것은 머리가 아팠다. (안될 건 없는데, 설마 구현을 깡으로 시킨다고...? 의 느낌)

그래서 답지를 봤다.

방법은 두 개의 자료구조를 사용하는 것이다.

하나의 더블 링크드 리스트와 이를 찾을 수 있도록 하는 map 하나이다.

삽입이 되면 뒤로 넣어주고, 가득 차 있었다면 앞에 있는 애를 뺀다. map도 갱신한다.

찾기를 하면 map에서 iter를 찾아준 다음 이를 통해 값을 얻고, 이후 list를 갱신한다.

아래는 코드이다.

class LRUCache {

private:

    int size;

    list<pair<int, int>> data;

    unordered_map<int, list<pair<int, int>>::iterator> m;

public:

    LRUCache(int capacity) {

        size = capacity;

    }

    int get(int key) {

        if (m.find(key) == m.end()) return -1;

  

        auto list_iter = m[key];

        int value = list_iter->second;

        data.erase(list_iter);

        data.push_back(make_pair(key, value));

  

        m[key] = prev(data.end());

        return value;

    }

    void put(int key, int value) {

  

        if (m.find(key) == m.end()) {

            if (data.size() >= size) {

                int oldest_key = data.front().first;

                m.erase(oldest_key);

                data.pop_front();

            }

            data.push_back(make_pair(key, value));

            m[key] = prev(data.end());

        }    

  

        else {

            auto list_iter = m[key];

            data.erase(list_iter);

            data.push_back(make_pair(key, value));

  

            m[key] = prev(data.end());

        }

    }

}
