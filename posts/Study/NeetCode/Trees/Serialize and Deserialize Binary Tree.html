Serialize and Deserialize Binary Tree

트리 인코딩 디코딩 하기

푸는 방법

preorder로 순회하면서 파싱할 수 있는 단서를 남기면 된다.

left 는 ()로, right 는 {}로 표현해서 풀었다.

다음은 코드이다. 

class Codec {
public:

    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        string result = "";
        if (root == nullptr)return result; // 예외처리

        result.append(to_string(root->val)); 

        result.append("(");
        result.append(serialize(root->left));
        result.append(")");

        result.append("{");
        result.append(serialize(root->right));
        result.append("}");

        return result;
    }

    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        if (data.empty()) return nullptr;

        queue<char> q;
        for (auto a : data) q.push(a);

        auto dfs = [&](auto& self) -> TreeNode* {
            if (q.empty()) return nullptr;

            string val_str = "";
            while (!q.empty() && (isdigit(q.front()) || q.front() == '-')) {
                val_str +=q.front();
                q.pop();
            }

            if (val_str.empty()) return nullptr;

            TreeNode* root = new TreeNode(stoi(val_str));

            if (!q.empty() && q.front() == '(') {
                q.pop();
                root->left = self(self);
                q.pop(); 
            }

            if (!q.empty() && q.front() == '{') {
                q.pop();
                root->right = self(self);
                q.pop(); 
            }

            return root;
        };

        return dfs(dfs);
    }
};

문제는 음수 처리와 닫는 괄호도 생각하기 때문에 코드가 길고 복잡하다는 점이다.

이것보다 더욱 단순하게 \#과 ,로만 풀 수도 있다.

자식이 없을 경우 \#으로 표현하고, 다음으로 넘어갈 때는 ,로 표현하는 것이다.

이렇게 하면 , 전까지는 숫자임이 보장되니까, queue에 넣을 필요없이 getline 함수를 사용하면 쉽게 풀 수 있다.

다음은 코드이다. 

class Codec {
public:

    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        if (!root) return "#,";
        return to_string(root->val) + "," +
            serialize(root->left) + serialize(root->right);
    }

    TreeNode* decode(stringstream& out) {
        string val;
        getline(out, val, ',');
        if (val == "#") return nullptr;
        TreeNode* root = new TreeNode(stoi(val));

        root->left = decode(out);
        root->right = decode(out);
        return root;
    }
    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        stringstream out(data);
        return decode(out);
    }
};
