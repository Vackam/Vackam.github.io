N Queens를 풀어라.

푸는 방법

백트래킹의 아주 정석적인 문제인 N-Queens가 나왔다.

푸는 방법은 하나하나 정직하게 체크하면서 가면 된다.

다음은 코드이다.

class Solution {
    vector<bool> cols, diag1, diag2;

    void dfs(int row, int n, vector<int>& tracking, vector<vector<string>>& result) {
        if (row == n) {
            result.push_back(make_board(tracking, n));
            return;
        }
        for (int col = 0; col < n; ++col) {
            if (cols[col] || diag1[row + col] || diag2[row - col + n - 1]) continue;

            cols[col] = diag1[row + col] = diag2[row - col + n - 1] = true;
            tracking.push_back(col);

            dfs(row + 1, n, tracking, result);

            tracking.pop_back();
            cols[col] = diag1[row + col] = diag2[row - col + n - 1] = false;
        }
    }

    vector<string> make_board(const vector<int>& tracking, int n) {
        vector<string> board;
        for (int col : tracking) {
            string row(n, '.');
            row[col] = 'Q';
            board.push_back(row);
        }
        return board;
    }

public:
    vector<vector<string>> solveNQueens(int n) {
        cols.assign(n, false);
        diag1.assign(2 * n - 1, false);
        diag2.assign(2 * n - 1, false);
        vector<int> tracking;
        vector<vector<string>> result;
        dfs(0, n, tracking, result);
        return result;
    }
};

대각선 판정을 아름답게 하는 법이 이 문제의 관건이다.

결론

너무나도 유명한 문제여서 글을 줄이도록 하겠다.
