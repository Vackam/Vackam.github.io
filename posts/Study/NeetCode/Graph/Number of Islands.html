BFS로 풀면 된다.

다음은 코드이다.

class Solution {
private:
    vector<int> dx = {1, -1, 0, 0};
    vector<int> dy = {0, 0, 1, -1};
public:
    bool is_possible(int nx, int ny, int m, int n) {
        return (nx < m && nx >= 0 && ny < n && ny >= 0);
    }
    int numIslands(vector<vector<char>>& grid) {
        int m, n;

        m = grid.size();
        n = grid[0].size();

        vector<vector<bool>> visited(m, vector<bool>(n, false));

        int result = 0;

        for (int i=0; i<m; ++i) {
            for (int j=0; j<n; ++j) {
                if (visited[i][j]) continue;

                if (grid[i][j] == '1') {
                    result++;
                    queue<pair<int, int>> q;
                    q.push({i, j});

                    visited[i][j] = true;

                    while (!q.empty()) {
                        auto a = q.front();
                        q.pop();

                        for (int i=0; i<4; ++i) {
                            int nx = a.first + dx[i];
                            int ny = a.second + dy[i];

                            if (is_possible(nx, ny, m, n) && !visited[nx][ny] && grid[nx][ny] == '1') {
                                q.push({nx, ny});
                                visited[nx][ny] = true;
                            }
                        }
                    }
                }

                else visited[i][j] = true;
            }
        }    

        return result;
    }
};
