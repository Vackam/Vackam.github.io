핸드폰 번호가 주어졌을 때, 가능한 경우를 구하시오

푸는 방법

단순하게 문자열을 저장하는 array를 하나 세우고 이를 만날 때마다 탐색하면 된다.

가지치기도 필요없이 정직하게 백트래킹으로 풀면된다.

다음은 코드이다.

class Solution {
    static constexpr array<string_view, 8> keypad = {"abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};
public:
    void dfs (const string& d, int now, string& path, vector<string>& result) {
        if (now == d.size()) {
            result.push_back(path);
            return;
        }

        // 올바른 digit인 지는 알아서 밖에서 체크해가지고 오는 걸로
        int target = d[now] - '2';
        for (int i=0; i<keypad[target].size(); ++i) {
            path.push_back(keypad[target][i]);
            dfs(d, now+1, path, result); 
            path.pop_back();
        }

        return;
    }
    vector<string> letterCombinations(string digits) {
        vector<string> result;
        if (digits.empty()) return result;
        string path = "";

        dfs(digits, 0, path, result);
        return result; 
    }
};
