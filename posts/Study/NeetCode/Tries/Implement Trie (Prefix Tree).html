쉽게 풀기 vs 어렵게 풀기가 있는데, 전자는 한 글자씩 찾아나가는 방식이고 후자는 공통되는 부분을 최대로 묶어놓는 방식이다.

이번에는 어렵게 풀어보자.

app -> apple -> apply 순으로 입력이 들어오면

첫번째로 app가 구성되고, 이후 app 아래 노드로 le를 가진 노드가 들어오고, apply가 들어오면 l에서 쪼개져서 e와 y가 결정되는 형식이다.

아래는 코드이다.

class PrefixTree {
private:
    struct RadixNode {
        map<char, pair<string, RadixNode*>> children;
        bool isEndOfWord = false;
    };

    RadixNode* root;
public:
    PrefixTree() {
        root = new RadixNode(); 
    }
    
    void insert(string word) {
        RadixNode* curr = root;
        int i = 0;

        while (i < word.length()) {
            char first_char = word[i];

            if (curr->children.find(first_char) == curr->children.end()) { // 첫글자가 매칭 안되면 새로 만들기.
                RadixNode* new_Node = new RadixNode();
                new_Node->isEndOfWord = true;
                curr->children[first_char] = {word.substr(i), new_Node}; 
            }

            auto& edge = curr->children[first_char];
            string edge_str = edge.first;
            RadixNode* next_Node = edge.second;

            int j = 0;
            while (j < edge_str.length() && 
                    i+j < word.length() &&
                    edge_str[j] == word[i+j]) j++; // 같은 부분 스킵
            if (j < edge_str.length()) {
                RadixNode* split_Node = new RadixNode();
                split_Node->children[edge_str[j]] = {edge_str.substr(j), next_Node};
                curr->children[first_char] = {edge_str.substr(0, j), split_Node};
                curr = split_Node; // 나누고 나서는 우리가 주목하고자 하는 곳에 이동
            }
            else curr = next_Node; // 전부 같다면 다음 노드로 이동
            i += j; // 덧셈하기
        }  
        curr->isEndOfWord = true; // 끝났으면 단어의 끝이라고 알려주기
    }
    
    bool search(string word) {
        RadixNode* curr = root;
        int i = 0;

        while (i < word.length()) {
            char first_char = word[i];
            if (curr->children.find(first_char) == curr->children.end()) {
                return false;
            }

            auto& edge = curr->children[first_char];
            string edge_str = edge.first;
            
            if (word.substr(i, edge_str.length()) != edge_str) {
                return false;
            }
            
            i += edge_str.length();
            curr = edge.second;
        } 
        return curr->isEndOfWord;
    }
    
    bool startsWith(string prefix) {
        RadixNode* curr = root;
        int i = 0;
        
        while (i < prefix.length()) {
            char first_char = prefix[i];
            if (curr->children.find(first_char) == curr->children.end()) {
                return false;
            }
            
            auto& edge = curr->children[first_char];
            string edge_str = edge.first;
            
            // prefix의 남은 부분이 edge_str보다 짧을 수 있으므로 공통 비교
            int compare_len = min(edge_str.length(), prefix.length() - i);
            if (prefix.substr(i, compare_len) != edge_str.substr(0, compare_len)) {
                return false;
            }
            
            i += compare_len;
            curr = edge.second;
        }
        
        return true;
    }
};

묶고 쪼개는 과정이 추가되어 아주 복잡해졌지만, 효율성이 좋다.

결론

점점 깡구현을 못해가는 몸이 되가는 것 같다. 사실 문제보자마자 못 풀겠다 싶어서 바로 도움을 받았다. 좀 더 고민하는 습관을 가지자.
