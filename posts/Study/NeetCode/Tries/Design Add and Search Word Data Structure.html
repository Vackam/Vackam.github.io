Design Add and Search Word Data Structure

2026-07-11 ps트리

서론

| https://neetcode.io/problems/design-word-search-data-structure/question

모든 문자를 의미하는 .을 포함하여 trie를 구성하라

푸는 방법

Implement Trie (Prefix Tree) 여기서 조금 변형하여 풀면 된다.

다만, 위 방법에서는 Radix Tree를 구성했는데 이번 문제에서는 .을 처리하기 위해서 단순한 방법으로 Trie를 구현했다.

queue를 이용하여 path를 탐색하도록 했고, .을 만나게 되면 해당 노드가 가지고 있는 모든 path를 탐색하도록 설정하였다.

다음은 코드이다. 

class WordDictionary {
private:
    struct TreeNode {
        vector<TreeNode*> v;
        bool is_end;
        TreeNode() : v(26, nullptr), is_end(false) {}
    };

    TreeNode* root;
public:
    WordDictionary() {
        root = new TreeNode(); 
    }
    
    void addWord(string word) {
        int i = 0;
        int size = word.size();
        TreeNode* now_node = root;
        for (i; i<size; ++i) {
            if (now_node->v[alpha_to_int(word[i])]) {
                // 존재할 때
                now_node = now_node->v[alpha_to_int(word[i])];
            }
            else {
                // 존재하지 않을 때
                break;
            }
        }

        if (i == size) {
            // 중복 포함 찾았을 때 그냥 treu 화 하기
            now_node->is_end = true;
        }

        else {
            // 전체 탐색하지 못하고 계속 되었을 떄
            for (i; i<size; ++i) {
                TreeNode* new_node = new TreeNode();
                now_node->v[alpha_to_int(word[i])] = new_node;
                now_node = new_node;  
            }
            now_node->is_end = true;
        }
    }
    
    bool search(string word) {
        int size = word.size();
        queue<pair<int, TreeNode*>> q; // 자기가 봐야 하는 것 + 순번

        q.push({0, root});

        while (!q.empty()) {
            auto now = q.front();
            q.pop();

            if (now.first == size) {
                if (now.second->is_end) return true; // 동일하고 끝이라면 성공
                continue; // 아니면 이 경로 아님.
            }

            if (word[now.first] == '.') {
                // .일 경우는 그 라딕스 노드에 있는 모든 가능한 놈들 다 넣기
                for (auto& n : now.second->v) {
                    if (n != nullptr) {
                        q.push({now.first + 1, n});
                    }
                }
            }
            else {
                // .이 아닌 경우는 탐색
                if (now.second->v[alpha_to_int(word[now.first])] != nullptr) {
                    q.push({now.first + 1, now.second->v[alpha_to_int(word[now.first])]});
                }
            }
        }

        return false; // 못찾음.
    }

    int alpha_to_int(char c) { return c - 'a'; }
};
