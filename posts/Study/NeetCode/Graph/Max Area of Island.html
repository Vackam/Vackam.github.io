Number of Islands

위 문제에서 섬 크기를 세주는 코드만 작성하면 된다.

다음은 코드이다.

class Solution {
public:
    int maxAreaOfIsland(vector<vector<int>>& grid) {
        vector<int> dx = {1, -1, 0, 0};    
        vector<int> dy = {0, 0, 1, -1};    

        int m = grid.size();
        int n = grid[0].size();

        auto check = [&](int x, int y){
            return (x >= 0 && x < m && y >= 0 && y < n);
        };

        int result = 0;

        for (int i=0; i<m; ++i) {
            for (int j=0; j<n; ++j) {
                if (grid[i][j] == 0) continue;

                if (grid[i][j] == 1) {
                    int size = 0;
                    queue<pair<int, int>> q;
                    q.push({i, j});
                    grid[i][j] = 0;

                    while (!q.empty()) {
                        size++;

                        auto a = q.front();
                        q.pop();

                        for (int k=0; k<4; ++k) {
                            int nx = a.first + dx[k];
                            int ny = a.second + dy[k];
                            if (check(nx, ny) && grid[nx][ny] == 1) {
                                q.push({nx, ny});
                                grid[nx][ny] = 0;
                            } 
                        }
                    }

                    if (size > result) result = size;
                }
            }
        }

        return result;
    }
};
