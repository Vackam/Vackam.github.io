푸는 방법

Word Search II

이거 풀기 전에 풀었어야 하는 문제인 듯 하다.

저 문제에선 Trie를 해서 풀었어야 했지만, 한 단어만 찾는 것이면 단순하게 백트래킹으로 풀 수 있다.

시작점을 찾고 이를 계속 dfs로 탐색해주면 된다.

항상 사용하는 4방향 연산을 적용해주면 된다.

다음은 코드이다.

class Solution {
    vector<int> dx = {1, -1, 0, 0};
    vector<int> dy = {0, 0, 1, -1};
public:
    int dfs (vector<vector<char>>& board, string word, int now, int x, int y, int n, int m) {
        // now 는 이제 +1로 들어온 친구. 즉 탐색해야 함.
        // 탐색까지 왔으면 참
        if (now == word.size()) return 1;

        int result = 0;
        
        for (int i=0; i<4; ++i) {
            int nx = x + dx[i];
            int ny = y + dy[i];

            if (nx < 0 || nx >= n ||
                ny < 0 || ny >= m) continue; // 범위 밖

            if (board[nx][ny] == word[now]) {
                // 이때만 탐색
                // 탐색한 곳 visited 처리
                char temp = board[nx][ny];
                board[nx][ny] = '#';
                result += dfs(board, word, now + 1, nx, ny, n, m); 
                board[nx][ny] = temp;
            } 
        }

        return result;
    }
    bool exist(vector<vector<char>>& board, string word) {
        // 찾고 dfs 조지고 찾고 하면 될 거 같은데
        int n = board.size();
        int m = board[0].size();

        for (int i=0; i<n; ++i) {
            for (int j=0; j<m; ++j) {
                if (word[0] == board[i][j]) {
                    // dfs 조지기 
                    board[i][j] = '#';
                    if (dfs(board, word, 1, i, j, n, m) > 0) return true;
                    board[i][j] = word[0];
                }
            }
        }

        return false;
    }
};
